Language — Functions

Language — Functions

This page covers how to define and use functions in Quill scripts, both custom functions and built-in ones. Functions let you organize code into reusable blocks and break down complex logic into smaller pieces.

Defining Functions

Create custom functions using the func keyword:

func function_name(parameter1, parameter2) {
    // function body
    return value;  // optional
}

Functions can:
- Take any number of parameters
- Return values (optional)
- Access global scope variables
- Call other functions

Example function definitions:

// Basic function with return
func add(a, b) {
    return a + b;
}

// Void function (no return)
func heal_team(team_scope) {
    for player in team_scope.players {
        heal(player);
    }
}

// Multiple parameters
func give_with_message(player, item_id, amount) {
    if give(player, item_id, amount) {
        sendmessage(player, "Given {amount}x {item_id}");
        return true;
    }
    return false;
}

Calling Functions

Call functions by name with parentheses:

// Call and use return value
let sum = add(5, 3);

// Call without using return
heal_team(red_team);

// Store result for error checking
let success = give_with_message(player, "diamond", 64);
if !success {
    log("Failed to give diamonds");
}

Parameter Rules

  • Parameters can be any type
  • Arguments are passed by value for primitives
  • Complex types (Player, Scope, etc.) are passed by reference
  • Default values are not supported (use multiple functions instead)
// Example showing parameter behavior
func modify(number, list, player) {
    number = number + 1;        // Only affects local copy
    append(list, "new");        // Modifies original list
    heal(player);              // Affects actual player
    player = null;             // Only affects local reference
}

let num = 5;
let items = ["old"];
let p = get_player("Steve");

modify(num, items, p);
// num is still 5
// items is now ["old", "new"]
// p is still healed but not null

Return Values

Functions can return any type or nothing:

// Return different types
func get_value(type) {
    if type == "number" {
        return 42;
    } else if type == "string" {
        return "hello";
    } else if type == "list" {
        return [1, 2, 3];
    }
    return null;  // Default return
}

// Early returns for validation
func safe_teleport(player, x, y, z) {
    if !player {
        return false;
    }

    if y < 0 || y > 255 {
        return false;
    }

    return teleport(player, x, y, z);
}

Scope Access

Functions can access the global scope and create local variables:

// Global variable
let total_healings = 0;

func heal_if_needed(player) {
    // Local variable
    let current_health = get_health(player);

    if current_health < 20 {
        heal(player);
        // Access global variable
        total_healings = total_healings + 1;
        return true;
    }
    return false;
}

Built-in Functions

Quill provides many built-in functions categorized by purpose:

Player Functions

Essential player interactions:

teleport(player, x, y, z)
give(player, item_id, amount)
heal(player)
kill(player)
sendmessage(player, message)

World Functions

Modify the environment:

set_block(x, y, z, block_id)
spawn_entity(entity_type, location)
create_explosion(x, y, z, power)
set_time(world, time)

Scope Functions

Manage player regions:

in_region(player, scope)
addtoscope(player, scope)
getplayers(scope)

Utility Functions

Helper functions:

wait(ticks)
random(min, max)
len(list)
type_of(value)
log(message)

Best Practices

  1. Meaningful Names
    ```quill
    // Good
    func calculate_damage(weapon, armor)

// Bad
func calc(w, a)
```

  1. Single Responsibility
    ```quill
    // Good
    func check_health(player) {
    return get_health(player) > 10;
    }

func heal_if_low(player) {
if !check_health(player) {
heal(player);
}
}

// Bad
func do_health_stuff(player) {
// Too many responsibilities
if get_health(player) > 10 {
give(player, "golden_apple");
} else {
heal(player);
sendmessage(player, "Healed!");
}
}
```

  1. Validate Parameters
    ```quill
    func give_items(player, amount) {
    // Check parameters first
    if !player || !is_online(player) {
    return false;
    }

    if amount < 1 {
    return false;
    }

    // Main logic after validation
    return give(player, "diamond", amount);
    }
    ```

  2. Return Consistent Types
    ```quill
    // Good - always returns number
    func get_max_health(entity_type) {
    if entity_type == "zombie" {
    return 20;
    }
    return 10; // Default
    }

// Bad - inconsistent returns
func get_health_info(entity_type) {
if entity_type == "zombie" {
return 20;
}
return "unknown"; // Different type
}
```

Examples

Complex game function using multiple concepts:

let arena = new Scope(-50, 60, -50, 50, 100, 50);

func start_game(required_players) {
    // Validation
    if len(arena.players) < required_players {
        broadcast("Not enough players!");
        return false;
    }

    // Setup
    arena.game_active = true;
    arena.remaining_players = len(arena.players);

    // Player initialization
    for player in arena.players {
        reset_player(player);
    }

    broadcast("Game started with {arena.remaining_players} players!");
    return true;
}

func reset_player(player) {
    // Health and game mode
    heal(player);
    clear_effects(player);
    set_gamemode(player, "survival");

    // Equipment
    give(player, "minecraft:iron_sword");
    give(player, "minecraft:bread", 5);

    // Random spawn within arena
    let x = random(-45, 45);
    let z = random(-45, 45);
    teleport(player, x, 65, z);
}

func eliminate_player(player) {
    if !arena.game_active {
        return false;
    }

    set_gamemode(player, "spectator");
    arena.remaining_players = arena.remaining_players - 1;
    broadcast("{player.name} was eliminated!");

    // Check win condition
    if arena.remaining_players <= 1 {
        end_game();
    }

    return true;
}
