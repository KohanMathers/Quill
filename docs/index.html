// Create an arena mini-game
let arena = new Scope(-50, 60, -50, 50, 100, 50);
arena.game_active = false;

// Start game on chat command
OnEvent(PlayerChat) {
    if chat.message == "!start" {
        arena.game_active = true;
        broadcast("Game starting!");
        
        for player in arena.players {
            set_gamemode(player, "survival");
            give(player, "minecraft:stone_sword");
            teleport(player, 0, 65, 0);
        }
    }
}

// Handle player deaths
OnEvent(PlayerDeath) {
    if arena.game_active && in_region(player, arena) {
        broadcast("{player.name} was eliminated!");
    }
}

Getting Started

Installation

Requirements

  • Minecraft Server (Bukkit/Spigot/Paper)
  • Java 21 or higher

Setup

  1. Download the Quill-0.1.0-ALPHA.jar from Releases
  2. Place in your server's plugins/ folder
  3. Restart your server
  4. Configure in plugins/Quill/config.yml
/quill scope create <scope_name> <owner> <x1> <y1> <z1> <x2> <y2> <z2>

Create a scope for a player to start scripting

// Example: Teleport pads
let pad1 = null;
let pad2 = null;

OnEvent(PlayerInteract) {
    if is_op(player) {
        let held_item = interact.item;
        
        if held_item != null {
            if held_item.type == "wooden_sword" {
                pad1 = interact.block;
                sendmessage(player, "pad 1 set at {pad1.x}, {pad1.y}, {pad1.z}");
            }
            
            if held_item.type == "stone_sword" {
                pad2 = interact.block;
                sendmessage(player, "pad 2 set at {pad2.x}, {pad2.y}, {pad2.z}");
            }
        }
    }
}

OnEvent(PlayerChat) {
    if chat.message == "!teleport" {
        let stand_loc1 = location(pad1.x, pad1.y + 1, pad1.z);
        let stand_loc2 = location(pad2.x, pad2.y + 1, pad2.z);
        let player_location = location(floor(player.location.x), floor(player.location.y), floor(player.location.z));
        if distance(player_location, stand_loc1) < 1 {
            teleport(player, stand_loc2);
        } else if distance(player_location, stand_loc2) < 1 {
            teleport(player, stand_loc1);
        }
    }
}
